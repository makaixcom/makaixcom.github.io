# 文件方式修改（省事）

# 步骤 1： 修改配置

打开 vi /etc/NetworkManager/system-connections/ens33.nmconnection 文件，并将 [ipv4] 的内容改为以下内容

[ipv4]
method=manual
dns=180.76.76.76
gateway=192.168.3.2
address=192.168.3.133/24

# 步骤 2： 应用设置并重启

先重新加载配置文件，再重启 ens33 网卡

nmcli c reload # 重新加载配置文件
nmcli c up ens33  # 重启ens33网卡

# 命令方式修改（上面配置完这个忽略）

# 步骤 1： 查看

nmcli device

# 步骤 2： 起个别名方便操作（可以忽略）

alias cm="nmcli connection modify"

# 步骤 3： 设置 ipv4 相关信息

cm ens33 ipv4.addresses 192.168.3.133/24
cm ens33 ipv4.method manual
#cm ens33 ipv4.gateway, .dns, ...    剩下的自己配吧

# 步骤 4： 应用设置并重启

nmcli connection down ens33 ; nmcli connection up ens33
